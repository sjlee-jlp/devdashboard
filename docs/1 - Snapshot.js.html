/**
 * Creates a 'snapshot' of the Current Week sheet, placing the inforamtion
 * into the Archive sheet and adding a datestamp
 *
 * @property {snapshotArray[]} snapshotArray Used to store the values in the Current Week sheet
 * @property {object} cwSheet The Spreadsheet for the Current Week
 * @property {object} archiveSheet The Spreadsheet used for archiving
 * @property {object} lock Applies a script lock to stop simultaneous running
 * @throws Will throw an error if there is any issue. Stops the script from running.
 * @name clearContent_
 * @function
 */
function snapshot_() {
  var snapshotArray = [];
  if (cwSheet.getLastRow() > 1) {
    var lastRow = cwSheet.getLastRow();
    var lastCol = cwSheet.getLastColumn();
    /** Start at row 2 (avoiding headers), column 1, for lastRow-1 rows (they are 0 indexed so need the -1) and lastCol columns (not 0 indexed)*/
    var cwRange = cwSheet.getRange(2, 1, lastRow - 1, lastCol);
    var cwData = cwRange.getValues();
    if (cwData.length > 0) {
      var today = new Date();
      var lock = LockService.getScriptLock();

      try {
        /** For each row push the data to an array */
        cwData.forEach(function(row, i) {
          if (row[0] !== '') {
            snapshotArray.push(cwData[i]);
          }
        });
        /** If the array is not empty then add todays date to the begining of the array*/
        if (snapshotArray.length > 0) {
          snapshotArray.forEach(function(element) {
            element.unshift(today);
            snapshotArray.push.apply(element);
          });
          /** Apply the array to the archive sheet*/
          archiveSheet
            .getRange(
              archiveSheet.getLastRow() + 1,
              1,
              snapshotArray.length,
              lastCol + 1
            )
            .setValues(snapshotArray);
          /** Make sure the changes to the spreadsheet are compelted*/
          SpreadsheetApp.flush();
          /** Clears the content for the rows that had been copied over*/
          //          clearContent_();
        }
      } catch (err) {
        throw err;
      }
      lock.releaseLock();
      if (lock.hasLock()) {
        throw new Error('Lock violation');
      }
    }
  }
}
